#include <iostream>
#include <string>
using namespace std; 	 

struct Node 
{
   string data;
   Node *next;        
};
typedef Node * NodeP;

int main(void)
{
   //head node
   NodeP head(new Node);
   head->next = NULL;
   head->data = "me";
   
   //first node
   NodeP first(new Node);
   first->next = NULL;
   first->data = "you";
   
   //link two nodes
   head->next = first;
   
   //print the whole chain
   NodeP tmp = head;
   while (tmp != NULL)
   {
     cout << tmp->data << endl;
     tmp = tmp->next;
   }
   
   return 0;
}
